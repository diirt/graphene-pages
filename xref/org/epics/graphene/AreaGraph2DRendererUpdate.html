5   package org.epics.graphene;
6   
7   /**
8    * A set of parameters that can be applied to a <code>AreaGraph2DRenderer</code>
9    * to update it's settings.
10   * 
11   * <p>
12   * Only the parameters that are set in the update get applied to the renderer.
13   * Parameters unique to the area graph that can be changed are:
14   * <ul>
15   *          <b>highlightFocusValue</b>
16   *          <b>focusPixelX</b>
17   * </ul>
18   * 
19   * @author carcassi
20   */
21  public class AreaGraph2DRendererUpdate extends Graph2DRendererUpdate<AreaGraph2DRendererUpdate> {
22      
23      private Integer focusPixelX;
24      private Boolean highlightFocusValue;
25      
26      public AreaGraph2DRendererUpdate focusPixel(int x) {
27          this.focusPixelX = x;
28          return this;
29      }
30      
31      public AreaGraph2DRendererUpdate highlightFocusValue(boolean highlightFocusValue) {
32          this.highlightFocusValue = highlightFocusValue;
33          return this;
34      }
35      
36      public Integer getFocusPixelX() {
37          return focusPixelX;
38      }
39  
40      public Boolean getHighlightFocusValue() {
41          return highlightFocusValue;
42      }
43  }
