5   package org.epics.graphene;
6   
7   /**
8    * The scale to be used to create axis references and rescale values.
9    *
10   * @author carcassi
11   */
12  public interface ValueScale {
13      
14      /**
15       * Scales the value from range A to range B.
16       * @param value original value in original range
17       * @param minValue original range min
18       * @param maxValue original range max
19       * @param newMinValue new range min
20       * @param newMaxValue new range max
21       * @return new value in new range
22       */
23      double scaleValue(double value, double minValue, double maxValue, double newMinValue, double newMaxValue);
24      
25      /**
26       * Returns the reference axes for a given range
27       * @param range the range of values to create a reference axis for
28       * @param minRefs minimum references
29       * @param maxRegs max references
30       * @return reference axes
31       */
32      ValueAxis references(Range range, int minRefs, int maxRegs);
33  }
