21      @Test
22      public void range1() throws Exception {
23          Range range = RangeUtil.range(0.0, 10.0);
24          assertThat(range.getMinimum(), equalTo((Number) 0.0));
25          assertThat(range.getMaximum(), equalTo((Number) 10.0));
26      }
27      
28      public void range2() throws Exception {

32      }

69      @Test
70      public void subrange1() {
71          assertThat(RangeUtil.contains(RangeUtil.range(0.0, 1.0), RangeUtil.range(0.5, 0.75)), equalTo(true));
72          assertThat(RangeUtil.contains(RangeUtil.range(0.0, 1.0), RangeUtil.range(0.5, 1.0)), equalTo(true));
73          assertThat(RangeUtil.contains(RangeUtil.range(0.0, 1.0), RangeUtil.range(0.0, 0.75)), equalTo(true));
74          assertThat(RangeUtil.contains(RangeUtil.range(0.0, 1.0), RangeUtil.range(-1.0, 0.75)), equalTo(false));
75          assertThat(RangeUtil.contains(RangeUtil.range(0.0, 1.0), RangeUtil.range(0.0, 1.75)), equalTo(false));
76      }
77      
78      @Test
79      public void sum1() {
80          Range range1 = RangeUtil.range(0.0, 5.0);
81          Range range2 = RangeUtil.range(1.0, 2.0);
82          assertThat(RangeUtil.sum(range1, range2), sameInstance(range1));
83          assertThat(RangeUtil.sum(range2, range1), sameInstance(range1));
84      }
85      
86      @Test
87      public void sum2() {
88          Range range1 = RangeUtil.range(0.0, 5.0);
89          Range range2 = RangeUtil.range(1.0, 6.0);
90          Range range = RangeUtil.sum(range1, range2);
91          assertThat(range.getMinimum(), equalTo((Number) 0.0));
92          assertThat(range.getMaximum(), equalTo((Number) 6.0));
93          range = RangeUtil.sum(range2, range1);
94          assertThat(range.getMinimum(), equalTo((Number) 0.0));
95          assertThat(range.getMaximum(), equalTo((Number) 6.0));
96      }
