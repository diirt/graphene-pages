5   package org.epics.graphene;

7   import org.epics.util.stats.Range;
8   import org.epics.util.stats.Ranges;
9   import org.junit.Test;
10  import static org.junit.Assert.*;
11  import static org.hamcrest.Matchers.*;
12  
13  /**
14   *
15   * @author carcassi
16   */
17  public class AxisRangesTest {
18      
19      public AxisRangesTest() {
20      }
21  
22      @Test
23      public void absolute1() {
24          AxisRangeInstance axisRange = AxisRanges.absolute(0.0, 10.0).createInstance();
25          Range range = axisRange.axisRange(Ranges.range(3.0, 15.0), Ranges.range(-3.0, 4.0));
26          assertThat(range.getMinimum(), equalTo((Number) 0.0));
27          assertThat(range.getMaximum(), equalTo((Number) 10.0));
28      }
29  
30      @Test(expected=IllegalArgumentException.class)
31      public void absolute2() {
32          AxisRange axisRange = AxisRanges.absolute(10.0, 0.0);
33      }
34  
35      @Test
36      public void relative1() {
37          AxisRangeInstance axisRange = AxisRanges.data().createInstance();
38          Range range = axisRange.axisRange(Ranges.range(3.0, 15.0), Ranges.range(-3.0, 4.0));
39          assertThat(range.getMinimum(), equalTo((Number) 3.0));
40          assertThat(range.getMaximum(), equalTo((Number) 15.0));
41  
42          range = axisRange.axisRange(Ranges.range(1.0, 5.0), Ranges.range(-3.0, 4.0));
43          assertThat(range.getMinimum(), equalTo((Number) 1.0));
44          assertThat(range.getMaximum(), equalTo((Number) 5.0));
45      }
46  
47      @Test
48      public void integrated1() {
49          AxisRangeInstance axisRange = AxisRanges.integrated().createInstance();
50          Range range = axisRange.axisRange(Ranges.range(3.0, 5.0), Ranges.range(-3.0, 4.0));
51          assertThat(range.getMinimum(), equalTo((Number) 3.0));
52          assertThat(range.getMaximum(), equalTo((Number) 5.0));
53  
54          range = axisRange.axisRange(Ranges.range(4.0, 15.0), Ranges.range(-3.0, 4.0));
55          assertThat(range.getMinimum(), equalTo((Number) 3.0));
56          assertThat(range.getMaximum(), equalTo((Number) 15.0));
57      }
58  
59      @Test
60      public void integrated2() {
61          AxisRangeInstance axisRange = AxisRanges.integrated().createInstance();
62          Range range = axisRange.axisRange(Ranges.range(3.0, 5.0), Ranges.range(-3.0, 4.0));
63          assertThat(range.getMinimum(), equalTo((Number) 3.0));
64          assertThat(range.getMaximum(), equalTo((Number) 5.0));
65  
66          range = axisRange.axisRange(Ranges.range(1000000.0, 1000015.0), Ranges.range(-3.0, 4.0));
67          assertThat(range.getMinimum(), equalTo((Number) 1000000.0));
68          assertThat(range.getMaximum(), equalTo((Number) 1000015.0));
69      }
70  
71      @Test
72      public void display1() {
73          AxisRangeInstance axisRange = AxisRanges.display().createInstance();
74          Range range = axisRange.axisRange(Ranges.range(3.0, 15.0), Ranges.range(-3.0, 4.0));
75          assertThat(range.getMinimum(), equalTo((Number) (-3.0)));
76          assertThat(range.getMaximum(), equalTo((Number) 4.0));
77      }
78  
79      @Test
80      public void display2() {
81          AxisRangeInstance axisRange = AxisRanges.display().createInstance();
82          Range range = axisRange.axisRange(Ranges.range(3.0, 15.0), Ranges.range(0.0, 0.0));
83          assertThat(range.getMinimum(), equalTo((Number) 3.0));
84          assertThat(range.getMaximum(), equalTo((Number) 15.0));
85      }
86  }
