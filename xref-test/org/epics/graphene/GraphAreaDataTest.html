18      public void linearGraphRange() throws Exception {

20          GraphAreaData data = new GraphAreaData();
21          data.setGraphArea(5, 194, 294, 5);
22          data.setGraphPadding(0, 0, 0, 0);
23          data.setGraphBuffer(graphBuffer);
24          data.setLabelMargin(5, 5);
25          data.setRanges(Ranges.range(-5, 5), ValueScales.linearScale(), Ranges.range(-10, 10), ValueScales.linearScale());
26          graphBuffer.drawBackground(Color.WHITE);
27          data.prepareLabels(FontUtil.getLiberationSansRegular(), Color.BLACK);
28          data.prepareGraphArea(true, Color.GRAY);
29          data.drawGraphArea();
30          
31          ImageAssert.compareImages("graphAreaData.linearGraphRange", graphBuffer.getImage());
