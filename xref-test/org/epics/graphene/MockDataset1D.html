7   import java.util.Random;
8   import org.epics.util.array.ArrayDouble;
9   import org.epics.util.array.CollectionDouble;
10  import org.epics.util.array.CollectionNumber;
11  import org.epics.util.array.IteratorDouble;

15   *
16   * @author carcassi

18  public class MockDataset1D implements Point1DDataset {
19      
20      private ListNumber values;
21      private double minValue = Double.POSITIVE_INFINITY;
22      private double maxValue = Double.NEGATIVE_INFINITY;
23      private Statistics statistics;
24  
25      public MockDataset1D(double[] data) {
26          values = new ArrayDouble(data);
27          statistics = StatisticsUtil.statisticsOf(values);
28      }
29  
30      @Override
31      public Statistics getStatistics() {
32          return statistics;
33      }
34  
35      @Override
36      public int getCount() {
37          return values.size();
38      }
39  
40      @Override
41      public ListNumber getValues() {
42          return values;
43      }
44      
45      public static Point1DDataset uniform(double minValue, double maxValue, int nSamples) {
46          return new MockDataset1D(RangeUtil.createBins(minValue, maxValue, nSamples));
47      }
48      
49      public static Point1DDataset gaussian(int nSamples) {
50          Random rand = new Random();
51          double[] values = new double[nSamples];
52          for (int i = 0; i < values.length; i++) {
53              values[i] = rand.nextGaussian();
54          }
55          return new MockDataset1D(values);
56      }
57  }
